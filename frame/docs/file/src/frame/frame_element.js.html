function htmlEncode(s) {
	if(typeof s !== 'string') return htmlEncode(''+s) || '';
	if( s.indexOf('<') !== -1 ) s = s.replace(/</g,'&lt;');
	if( s.indexOf('>') !== -1 ) s = s.replace(/>/g,'&gt;');
	return s;
}
